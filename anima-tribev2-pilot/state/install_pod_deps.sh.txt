#!/usr/bin/env bash
set -eo pipefail
ts() { date -u +"%Y-%m-%dT%H:%M:%SZ"; }
log() { echo "[$(ts)] [install] $*"; }

log "START — uname=$(uname -a)"
log "python=$(python3 --version 2>&1) torch=$(python3 -c 'import torch;print(torch.__version__)' 2>&1)"

cd /workspace/anima_pilot_t1

log "Installing minimal deps for tribev2 forward..."
pip3 install --no-cache-dir \
  "neuralset==0.0.2" \
  "neuraltrain==0.0.2" \
  "x_transformers==1.27.20" \
  "einops" \
  "pyyaml" \
  "huggingface_hub" \
  "transformers" \
  "gtts" \
  "langdetect" \
  "soundfile" \
  "Levenshtein" \
  "julius" \
  "moviepy" \
  "spacy" \
  "pydantic" \
  "pandas" \
  "tqdm" 2>&1 | tail -30
RC=$?
log "minimal deps install rc=$RC"

log "Installing tribev2 in editable mode..."
pip3 install --no-cache-dir -e ./ref/tribev2 2>&1 | tail -30
RC=$?
log "tribev2 install rc=$RC"

log "Verifying TribeModel import..."
python3 -c "from tribev2.demo_utils import TribeModel; print('OK', TribeModel.__name__)" 2>&1
RC=$?
log "import rc=$RC"

log "Downloading spacy en_core_web_sm..."
python3 -m spacy download en_core_web_sm 2>&1 | tail -5 || log "WARN: spacy download failed"

log "Attempting whisperx install (non-fatal)..."
pip3 install --no-cache-dir whisperx 2>&1 | tail -5 || log "WARN: whisperx install failed"

log "DONE."
exit 0
