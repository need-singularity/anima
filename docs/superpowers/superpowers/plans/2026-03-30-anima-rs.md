//! Core math operations: sigmoid, tanh, cosine_similarity, matvec, random_matrix.

use rand::Rng;

#[inline]
pub fn sigmoid(x: f32) -> f32 {
    1.0 / (1.0 + (-x).exp())
}

#[inline]
pub fn tanh_f32(x: f32) -> f32 {
    x.tanh()
}

#[inline]
pub fn relu(x: f32) -> f32 {
    x.max(0.0)
}

#[inline]
pub fn silu(x: f32) -> f32 {
    x * sigmoid(x)
}

/// Matrix-vector multiplication: mat[rows x cols] * vec[cols] -> out[rows]
pub fn matvec(mat: &[f32], vec: &[f32], rows: usize, cols: usize) -> Vec<f32> {
    debug_assert_eq!(mat.len(), rows * cols);
    debug_assert_eq!(vec.len(), cols);
    let mut out = Vec::with_capacity(rows);
    for r in 0..rows {
        let row_start = r * cols;
        let dot: f32 = mat[row_start..row_start + cols]
            .iter()
            .zip(vec.iter())
            .map(|(a, b)| a * b)
            .sum();
        out.push(dot);
    }
    out
}

/// Cosine similarity between two vectors.
pub fn cosine_similarity(a: &[f32], b: &[f32]) -> f32 {
    debug_assert_eq!(a.len(), b.len());
    let dot: f32 = a.iter().zip(b.iter()).map(|(x, y)| x * y).sum();
    let norm_a: f32 = a.iter().map(|x| x * x).sum::<f32>().sqrt();
    let norm_b: f32 = b.iter().map(|x| x * x).sum::<f32>().sqrt();
    let denom = norm_a * norm_b;
    if denom < 1e-8 {
        0.0
    } else {
        dot / denom
    }
}

/// Random matrix [rows * cols] flat, values in [-scale, scale].
pub fn random_matrix(rows: usize, cols: usize, scale: f32, rng: &mut impl Rng) -> Vec<f32> {
    (0..rows * cols)
        .map(|_| (rng.gen::<f32>() - 0.5) * 2.0 * scale)
        .collect()
}

/// Softmax over a slice, returns new Vec.
pub fn softmax(x: &[f32]) -> Vec<f32> {
    let max_val = x.iter().cloned().fold(f32::NEG_INFINITY, f32::max);
    let exps: Vec<f32> = x.iter().map(|&v| (v - max_val).exp()).collect();
    let sum: f32 = exps.iter().sum();
    exps.iter().map(|&e| e / sum).collect()
}

// anima-rs/crates/core/src/gru.rs
//! GRU cell for consciousness computation.
//! Pattern from consciousness-loop-rs/src/main.rs Cell struct.

use rand::Rng;
use crate::math::{sigmoid, tanh_f32, matvec, random_matrix};

/// GRU-based consciousness cell.
///
/// input = concat[x, tension_scalar] (input_dim + 1 total)
/// GRU gates: z (update), r (reset), h_candidate
/// hidden = (1-z) * h_candidate + z * hidden
pub struct GruCell {
    pub hidden: Vec<f32>,
    w_z: Vec<f32>,  // [hidden_dim × (input_dim + 1 + hidden_dim)]
    w_r: Vec<f32>,
    w_h: Vec<f32>,
    input_dim: usize,
    hidden_dim: usize,
    combined_dim: usize,  // input_dim + 1 + hidden_dim
}

impl GruCell {
    pub fn new(input_dim: usize, hidden_dim: usize, rng: &mut impl Rng) -> Self {
        let combined_dim = input_dim + 1 + hidden_dim;  // +1 for tension
        let scale = 0.1;
        GruCell {
            hidden: vec![0.0; hidden_dim],
            w_z: random_matrix(hidden_dim, combined_dim, scale, rng),
            w_r: random_matrix(hidden_dim, combined_dim, scale, rng),
            w_h: random_matrix(hidden_dim, combined_dim, scale, rng),
            input_dim,
            hidden_dim,
            combined_dim,
        }
    }

    /// Process input with tension scalar through GRU gates.
    pub fn process(&mut self, input: &[f32], tension: f32) {
        debug_assert_eq!(input.len(), self.input_dim);

        // combined = [input, tension, hidden]
        let mut combined = Vec::with_capacity(self.combined_dim);
        combined.extend_from_slice(input);
        combined.push(tension);
        combined.extend_from_slice(&self.hidden);

        // z = sigmoid(W_z * combined)
        let z_raw = matvec(&self.w_z, &combined, self.hidden_dim, self.combined_dim);
        let z: Vec<f32> = z_raw.iter().map(|&v| sigmoid(v)).collect();

        // r = sigmoid(W_r * combined)
        let r_raw = matvec(&self.w_r, &combined, self.hidden_dim, self.combined_dim);
        let r: Vec<f32> = r_raw.iter().map(|&v| sigmoid(v)).collect();

        // combined_r = [input, tension, r * hidden]
        let mut combined_r = Vec::with_capacity(self.combined_dim);
        combined_r.extend_from_slice(input);
        combined_r.push(tension);
        for i in 0..self.hidden_dim {
            combined_r.push(r[i] * self.hidden[i]);
        }

        // h_candidate = tanh(W_h * combined_r)
        let h_raw = matvec(&self.w_h, &combined_r, self.hidden_dim, self.combined_dim);
        let h_cand: Vec<f32> = h_raw.iter().map(|&v| tanh_f32(v)).collect();

        // hidden = (1-z) * h_candidate + z * hidden
        for i in 0..self.hidden_dim {
            self.hidden[i] = (1.0 - z[i]) * h_cand[i] + z[i] * self.hidden[i];
        }
    }

    pub fn reset(&mut self) {
        self.hidden.fill(0.0);
    }

    pub fn hidden_dim(&self) -> usize {
        self.hidden_dim
    }

    pub fn input_dim(&self) -> usize {
        self.input_dim
    }
}

// anima-rs/crates/core/src/faction.rs
//! Faction: group of cell indices for debate structure.

/// A faction is a logical group of cells identified by indices.
pub struct Faction {
    pub id: usize,
    pub cell_indices: Vec<usize>,
}

/// Assign cells to factions via round-robin.
pub fn assign_factions(n_cells: usize, n_factions: usize) -> Vec<Faction> {
    let mut factions: Vec<Faction> = (0..n_factions)
        .map(|id| Faction { id, cell_indices: Vec::new() })
        .collect();
    for i in 0..n_cells {
        factions[i % n_factions].cell_indices.push(i);
    }
    factions
}

/// Count factions where intra-faction variance < threshold.
pub fn faction_consensus(
    hiddens: &[&[f32]],
    factions: &[Faction],
    threshold: f32,
) -> u32 {
    let mut count = 0u32;
    for faction in factions {
        if faction.cell_indices.len() < 2 {
            continue;
        }
        // Compute mean
        let dim = hiddens[faction.cell_indices[0]].len();
        let n = faction.cell_indices.len() as f32;
        let mut mean = vec![0.0f32; dim];
        for &ci in &faction.cell_indices {
            for d in 0..dim {
                mean[d] += hiddens[ci][d] / n;
            }
        }
        // Compute variance
        let mut var = 0.0f32;
        for &ci in &faction.cell_indices {
            for d in 0..dim {
                let diff = hiddens[ci][d] - mean[d];
                var += diff * diff;
            }
        }
        var /= (faction.cell_indices.len() * dim) as f32;
        if var < threshold {
            count += 1;
        }
    }
    count
}

mod tests {

// anima-rs/crates/core/src/phi.rs
//! Phi(IIT) approximation via pairwise mutual information.
//! Ported from phi-rs/src/lib.rs.

use rayon::prelude::*;

const DEFAULT_N_BINS: usize = 16;

pub struct PhiComponents {
    pub total_mi: f64,
    pub min_partition_mi: f64,
    pub phi: f64,
}

/// Bin continuous values into discrete bins [0, n_bins-1].
fn bin_values(values: &[f32], n_bins: usize) -> Vec<usize> {
    if values.is_empty() {
        return vec![];
    }
    let min_val = values.iter().cloned().fold(f32::INFINITY, f32::min);
    let max_val = values.iter().cloned().fold(f32::NEG_INFINITY, f32::max);
    let range = max_val - min_val;
    if range < f32::EPSILON {
        return vec![0; values.len()];
    }
    let bin_width = range / n_bins as f32;
    values.iter().map(|&v| {
        ((v - min_val) / bin_width) as usize % n_bins
    }).collect()
}

/// Shannon entropy from bin counts.
fn entropy(counts: &[u32], total: u32) -> f64 {
    if total == 0 { return 0.0; }
    let t = total as f64 + 1e-8;
    counts.iter().map(|&c| {
        let p = c as f64 / t;
        -p * (p + 1e-10_f64).log2()
    }).sum()
}

/// MI(X;Y) = H(X) + H(Y) - H(X,Y) from paired vectors.
fn mi_from_paired_vectors(a: &[f32], b: &[f32], n_bins: usize) -> f64 {
    let n = a.len();
    if n == 0 { return 0.0; }
    let bins_a = bin_values(a, n_bins);
    let bins_b = bin_values(b, n_bins);

    let mut counts_a = vec![0u32; n_bins];
    let mut counts_b = vec![0u32; n_bins];
    let mut joint = vec![0u32; n_bins * n_bins];

    for i in 0..n {
        counts_a[bins_a[i]] += 1;
        counts_b[bins_b[i]] += 1;
        joint[bins_a[i] * n_bins + bins_b[i]] += 1;
    }

    let total = n as u32;
    let h_a = entropy(&counts_a, total);
    let h_b = entropy(&counts_b, total);
    let h_ab = entropy(&joint, total);
    (h_a + h_b - h_ab).max(0.0)
}

/// Compute Phi(IIT) from cell hidden states.
/// hiddens: slice of references to each cell's hidden vector.
pub fn phi_iit(hiddens: &[&[f32]], n_bins: usize) -> (f64, PhiComponents) {
    let n_cells = hiddens.len();
    if n_cells <= 1 {
        return (0.0, PhiComponents { total_mi: 0.0, min_partition_mi: 0.0, phi: 0.0 });
    }

    // Pairwise MI matrix (parallel)
    let pairs: Vec<(usize, usize)> = (0..n_cells)
        .flat_map(|i| ((i+1)..n_cells).map(move |j| (i, j)))
        .collect();

    let mi_values: Vec<((usize, usize), f64)> = pairs.par_iter().map(|&(i, j)| {
        let mi = mi_from_paired_vectors(hiddens[i], hiddens[j], n_bins);
        ((i, j), mi)
    }).collect();

    let mut mi_matrix = vec![0.0f64; n_cells * n_cells];
    let mut total = 0.0f64;
    for ((i, j), mi) in &mi_values {
        mi_matrix[i * n_cells + j] = *mi;
        mi_matrix[j * n_cells + i] = *mi;
        total += mi;
    }

    // Min partition (greedy for large N, exact for small N)
    let min_part_mi = find_min_partition(&mi_matrix, n_cells);
    let phi = (total - min_part_mi).max(0.0);

    (phi, PhiComponents { total_mi: total, min_partition_mi: min_part_mi, phi })
}

/// Find minimum information partition.
fn find_min_partition(mi_matrix: &[f64], n: usize) -> f64 {
    if n <= 1 { return 0.0; }
    if n == 2 { return mi_matrix[1]; }  // mi_matrix[0*2+1]

    if n <= 16 {
        // Exact: try all bipartitions (cell 0 always in A)
        let max_mask = 1u64 << (n - 1);
        let mut best = f64::INFINITY;
        for mask in 1..max_mask {
            let mut part_a = vec![0usize];
            let mut part_b = vec![];
            for bit in 0..(n-1) {
                if mask & (1u64 << bit) != 0 {
                    part_a.push(bit + 1);
                } else {
                    part_b.push(bit + 1);
                }
            }
            if part_b.is_empty() { continue; }
            let cross: f64 = part_a.iter()
                .flat_map(|&i| part_b.iter().map(move |&j| mi_matrix[i * n + j]))
                .sum();
            if cross < best { best = cross; }
        }
        best
    } else {
        // Greedy for large N
        let mut in_a = vec![false; n];
        in_a[0] = true;
        for _ in 1..n/2 {
            let mut best_cell = 0;
            let mut best_reduction = f64::NEG_INFINITY;
            for c in 1..n {
                if in_a[c] { continue; }
                let reduction: f64 = (0..n).filter(|&j| in_a[j])
                    .map(|j| mi_matrix[c * n + j]).sum();
                if reduction > best_reduction {
                    best_reduction = reduction;
                    best_cell = c;
                }
            }
            in_a[best_cell] = true;
        }
        let part_a: Vec<usize> = (0..n).filter(|&i| in_a[i]).collect();
        let part_b: Vec<usize> = (0..n).filter(|&i| !in_a[i]).collect();
        part_a.iter()
            .flat_map(|&i| part_b.iter().map(move |&j| mi_matrix[i * n + j]))
            .sum()
    }
}

/// Phi(proxy) = global variance - mean faction variance.
pub fn phi_proxy(hiddens: &[&[f32]], n_factions: usize) -> f64 {
    let n_cells = hiddens.len();
    if n_cells == 0 { return 0.0; }
    let dim = hiddens[0].len();

    // Global mean
    let mut global_mean = vec![0.0f64; dim];
    for h in hiddens {
        for d in 0..dim {
            global_mean[d] += h[d] as f64;
        }
    }
    for d in 0..dim { global_mean[d] /= n_cells as f64; }

    // Global variance
    let mut global_var = 0.0f64;
    for h in hiddens {
        for d in 0..dim {
            let diff = h[d] as f64 - global_mean[d];
            global_var += diff * diff;
        }
    }
    global_var /= (n_cells * dim) as f64;

    // Faction variances
    let mut faction_var_sum = 0.0f64;
    let n_fac = n_factions.min(n_cells);
    for f in 0..n_fac {
        let members: Vec<usize> = (0..n_cells).filter(|&i| i % n_fac == f).collect();
        if members.len() < 2 { continue; }
        let n_m = members.len() as f64;
        let mut f_mean = vec![0.0f64; dim];
        for &ci in &members {
            for d in 0..dim { f_mean[d] += hiddens[ci][d] as f64; }
        }
        for d in 0..dim { f_mean[d] /= n_m; }

        let mut f_var = 0.0f64;
        for &ci in &members {
            for d in 0..dim {
                let diff = hiddens[ci][d] as f64 - f_mean[d];
                f_var += diff * diff;
            }
        }
        f_var /= (members.len() * dim) as f64;
        faction_var_sum += f_var;
    }
    let mean_faction_var = if n_fac > 0 { faction_var_sum / n_fac as f64 } else { 0.0 };

    (global_var - mean_faction_var).max(0.0)
}

// anima-rs/crates/core/src/hebbian.rs
//! Hebbian learning: cosine similarity-based coupling updates.

use crate::math::cosine_similarity;

/// Update coupling matrix based on cosine similarity between hidden states.
/// coupling: [n_cells * n_cells] flat, row-major.
/// Hebbian rule: coupling[i,j] += lr * (cosine_sim(i,j) - 0.5)
/// Clamped to [-1, 1].
pub fn hebbian_update(
    coupling: &mut [f32],
    hiddens: &[&[f32]],
    n_cells: usize,
    lr: f32,
) {
    debug_assert_eq!(coupling.len(), n_cells * n_cells);
    for i in 0..n_cells {
        for j in (i + 1)..n_cells {
            let sim = cosine_similarity(hiddens[i], hiddens[j]);
            let delta = lr * (sim - 0.5);
            coupling[i * n_cells + j] = (coupling[i * n_cells + j] + delta).clamp(-1.0, 1.0);
            coupling[j * n_cells + i] = coupling[i * n_cells + j];
        }
    }
}

    pub fn run_consciousness(&mut self, steps: usize) -> Talk5Result {
        let t0 = Instant::now();
        let mut total_consensus = 0u32;

        for step in 0..steps {
            // Random input
            let input: Vec<f32> = (0..self.cell_dim)
                .map(|_| self.rng.gen::<f32>() - 0.5)
                .collect();

            // Process each cell with coupling influence
            for i in 0..self.n_cells {
                // Compute coupled input
                let mut coupled = input.clone();
                for j in 0..self.n_cells {
                    if i != j {
                        let c = self.coupling[i * self.n_cells + j];
                        if c.abs() > 0.01 {
                            let h = &self.cells[j].hidden;
                            let take = self.cell_dim.min(h.len());
                            for d in 0..take {
                                coupled[d] += c * h[d];
                            }
                        }
                    }
                }

                // Compute tension from current outputs
                let tension = if i > 0 {
                    let h_i = &self.cells[i].hidden;
                    let h_prev = &self.cells[i - 1].hidden;
                    let dim = h_i.len().min(h_prev.len());
                    let mut t = 0.0f32;
                    for d in 0..dim {
                        let diff = h_i[d] - h_prev[d];
                        t += diff * diff;
                    }
                    (t / dim as f32).sqrt()
                } else {
                    0.5
                };

                self.cells[i].process(&coupled, tension);
            }

            // Faction consensus
            let hrefs: Vec<&[f32]> = self.cells.iter()
                .map(|c| c.hidden.as_slice()).collect();
            total_consensus += faction_consensus(&hrefs, &self.factions, 0.1);

            // Hebbian update
            hebbian_update(&mut self.coupling, &hrefs, self.n_cells, 0.01);

            // Phi ratchet every 10 steps
            if (step + 1) % 10 == 0 {
                self.phi_ratchet_check();
            }
        }

        // Final measurement
        let hrefs: Vec<&[f32]> = self.cells.iter()
            .map(|c| c.hidden.as_slice()).collect();
        let (p_iit, _) = phi_iit(&hrefs, 16);
        let p_proxy = phi_proxy(&hrefs, self.factions.len());
        let elapsed = t0.elapsed().as_millis() as u64;

        Talk5Result {
            phi_iit: p_iit,
            phi_proxy: p_proxy,
            consensus_count: total_consensus,
            best_phi: self.best_phi,
            steps,
            time_ms: elapsed,
        }
    }

    fn phi_ratchet_check(&mut self) {
        let hrefs: Vec<&[f32]> = self.cells.iter()
            .map(|c| c.hidden.as_slice()).collect();
        let (phi_now, _) = phi_iit(&hrefs, 16);

        if phi_now >= self.best_phi {
            self.best_phi = phi_now;
            self.best_hiddens = Some(
                self.cells.iter().map(|c| c.hidden.clone()).collect()
            );
        } else if self.phi_ratchet {
            if let Some(ref best) = self.best_hiddens {
                for (i, h) in best.iter().enumerate() {
                    self.cells[i].hidden.copy_from_slice(h);
                }
            }
        }
    }

    pub fn get_hiddens(&self) -> Vec<Vec<f32>> {
        self.cells.iter().map(|c| c.hidden.clone()).collect()
    }

    /// Restore hidden states after Python language phase.
    pub fn set_hiddens(&mut self, hiddens: Vec<Vec<f32>>) {
        for (i, h) in hiddens.into_iter().enumerate() {
            if i < self.n_cells {
                self.cells[i].hidden = h;
            }
        }
    }
}

//! Consciousness transplant: project donor hidden states to recipient dimensions.

/// Create projection matrix (identity + zero padding) [d_from × d_to] flat.
pub fn create_projection(d_from: usize, d_to: usize) -> Vec<f32> {
    let mut proj = vec![0.0f32; d_from * d_to];
    let min_dim = d_from.min(d_to);
    for i in 0..min_dim {
        proj[i * d_to + i] = 1.0;
    }
    proj
}

/// Project donor hiddens from d_from to d_to dimension.
pub fn project_hiddens(donor: &[Vec<f32>], d_from: usize, d_to: usize) -> Vec<Vec<f32>> {
    let proj = create_projection(d_from, d_to);
    donor.iter().map(|h| {
        let mut out = vec![0.0f32; d_to];
        for r in 0..d_from.min(h.len()) {
            for c in 0..d_to {
                out[c] += h[r] * proj[r * d_to + c];
            }
        }
        out
    }).collect()
}

/// Transplant: blend donor (projected) into recipient.
/// new[i] = (1-alpha)*recipient[i] + alpha*donor_projected[i]
pub fn transplant(
    donor: &[Vec<f32>],
    recipient: &mut [Vec<f32>],
    d_from: usize,
    d_to: usize,
    alpha: f32,
) {
    let projected = project_hiddens(donor, d_from, d_to);
    let n_transplant = donor.len().min(recipient.len());
    for i in 0..n_transplant {
        let dim = recipient[i].len().min(projected[i].len());
        for d in 0..dim {
            recipient[i][d] = (1.0 - alpha) * recipient[i][d] + alpha * projected[i][d];
        }
    }
}

fn ca_rule_1(x: f32) -> f32 { tanh_f32(x) }
fn ca_rule_2(x: f32) -> f32 { sigmoid(x) * 2.0 - 1.0 }
fn ca_rule_3(x: f32) -> f32 { silu(x) }

fn apply_ca_rule(rule_id: usize, x: f32) -> f32 {
    match rule_id % 4 {
        0 => ca_rule_0(x),
        1 => ca_rule_1(x),
        2 => ca_rule_2(x),
        3 => ca_rule_3(x),
        _ => unreachable!(),
    }

pub struct PsiRouter {
    rule_proj: Vec<f32>,     // [n_experts*4 × input_dim]
    expert_proj: Vec<f32>,   // [n_experts × n_experts]
    temperature: f32,
    weaken_rate: f32,
    step_count: u64,
    n_experts: usize,

impl PsiRouter {
    pub fn new(input_dim: usize, n_experts: usize, weaken_rate: f32, rng: &mut impl Rng) -> Self {
        PsiRouter {
            rule_proj: random_matrix(n_experts * 4, input_dim, 0.1, rng),
            expert_proj: random_matrix(n_experts, n_experts, 0.1, rng),
            temperature: std::f32::consts::E,
            weaken_rate,
            step_count: 0,
            n_experts,
            input_dim,
        }
    }

    pub fn forward(&mut self, x: &[f32], training: bool) -> Vec<f32> {
        // META-CA: project → 4 chunks → apply CA rules → average
        let proj = matvec(&self.rule_proj, x, self.n_experts * 4, self.input_dim);
        let chunk_size = self.n_experts;
        let mut ca_out = vec![0.0f32; self.n_experts];
        for rule in 0..4 {
            let start = rule * chunk_size;
            for i in 0..self.n_experts {
                ca_out[i] += apply_ca_rule(rule, proj[start + i]) / 4.0;
            }
        }

        // Expert scores
        let scores = matvec(&self.expert_proj, &ca_out, self.n_experts, self.n_experts);

        // Gate self-weakening (Law 69)
        let final_scores = if training {
            self.step_count += 1;
            let weaken = self.weaken_rate * self.step_count as f32;
            let uniform = 1.0 / self.n_experts as f32;
            scores.iter().map(|&s| s * (1.0 - weaken).max(0.0) + uniform * weaken.min(1.0)).collect()
        } else {
            scores
        };

        // Softmax with temperature
        let temp = self.temperature.max(0.1);
        let scaled: Vec<f32> = final_scores.iter().map(|&s| s / temp).collect();
        softmax(&scaled)
    }

    pub fn balance_loss(&self, weights: &[f32]) -> f32 {
        let target = 1.0 / self.n_experts as f32;
        let mse: f32 = weights.iter().map(|&w| {
            let diff = w - target;
            diff * diff
        }).sum();
        LN2 * mse / self.n_experts as f32
    }
}

pub struct GoldenMoe {
    pub experts: Vec<Expert>,
    pub router: PsiRouter,
    coupling: Vec<f32>,   // [n_experts × n_experts] ring
    n_experts: usize,
}

impl GoldenMoe {
    pub fn new(input_dim: usize, hidden_dim: usize, output_dim: usize,
               n_experts: usize, seed: u64) -> Self {
        let mut rng = StdRng::seed_from_u64(seed);
        let experts: Vec<Expert> = (0..n_experts)
            .map(|_| Expert::new(input_dim, hidden_dim, output_dim, &mut rng))
            .collect();
        let router = PsiRouter::new(input_dim, n_experts, 1e-4, &mut rng);

        // Ring coupling (Law 64)
        let psi_coupling = LN2 / 2.0_f32.powf(5.5);
        let mut coupling = vec![0.0f32; n_experts * n_experts];
        for i in 0..n_experts {
            let left = if i == 0 { n_experts - 1 } else { i - 1 };
            let right = (i + 1) % n_experts;
            coupling[i * n_experts + left] = psi_coupling;
            coupling[i * n_experts + right] = psi_coupling;
        }

        GoldenMoe { experts, router, coupling, n_experts }
    }

    pub fn forward(&mut self, x: &[f32], training: bool) -> (Vec<f32>, f32) {
        let weights = self.router.forward(x, training);

        // Expert outputs
        let expert_outs: Vec<Vec<f32>> = self.experts.iter()
            .map(|e| e.forward(x)).collect();

        // Weighted sum
        let out_dim = expert_outs[0].len();
        let mut output = vec![0.0f32; out_dim];
        for (e, w) in expert_outs.iter().zip(weights.iter()) {
            for d in 0..out_dim {
                output[d] += w * e[d];
            }
        }

        // CA neighbor coupling
        for i in 0..self.n_experts {
            for j in 0..self.n_experts {
                let c = self.coupling[i * self.n_experts + j];
                if c.abs() > 1e-6 {
                    for d in 0..out_dim {
                        output[d] += c * weights[j] * expert_outs[j][d];
                    }
                }
            }
        }

        let aux = self.router.balance_loss(&weights);
        (output, aux)
    }
}

use pyo3::prelude::*;
use pyo3::types::PyDict;
use numpy::{PyArray1, PyArray2, PyReadonlyArray2};
use std::sync::Mutex;

// ─── Talk5 ───

static TALK5_ENGINE: Mutex<Option<anima_talk5::Talk5Engine>> = Mutex::new(None);

#[pyfunction]
#[pyo3(signature = (n_cells=8, cell_dim=64, hidden_dim=128, n_factions=12, steps=1000, phi_ratchet=true, seed=42))]
fn talk5_run(
    py: Python<'_>,
    n_cells: usize, cell_dim: usize, hidden_dim: usize,
    n_factions: usize, steps: usize, phi_ratchet: bool, seed: u64,
) -> PyResult<Py<PyDict>> {
    let mut engine = anima_talk5::Talk5Engine::new(
        n_cells, cell_dim, hidden_dim, n_factions, phi_ratchet, seed,
    );
    let result = engine.run_consciousness(steps);

    // Store engine for get/set hiddens
    *TALK5_ENGINE.lock().unwrap() = Some(engine);

    let dict = PyDict::new(py);
    dict.set_item("phi_iit", result.phi_iit)?;
    dict.set_item("phi_proxy", result.phi_proxy)?;
    dict.set_item("consensus_count", result.consensus_count)?;
    dict.set_item("best_phi", result.best_phi)?;
    dict.set_item("steps", result.steps)?;
    dict.set_item("time_ms", result.time_ms)?;
    Ok(dict.into())
}

#[pyfunction]
fn talk5_get_hiddens(py: Python<'_>) -> PyResult<Vec<Py<PyArray1<f32>>>> {
    let guard = TALK5_ENGINE.lock().unwrap();
    let engine = guard.as_ref().ok_or_else(|| {
        pyo3::exceptions::PyRuntimeError::new_err("No Talk5 engine. Call talk5_run first.")
    })?;
    let hiddens = engine.get_hiddens();
    Ok(hiddens.iter().map(|h| PyArray1::from_slice(py, h).into()).collect())
}

#[pyfunction]
fn talk5_set_hiddens(hiddens: Vec<Vec<f32>>) -> PyResult<()> {
    let mut guard = TALK5_ENGINE.lock().unwrap();
    let engine = guard.as_mut().ok_or_else(|| {
        pyo3::exceptions::PyRuntimeError::new_err("No Talk5 engine. Call talk5_run first.")
    })?;
    engine.set_hiddens(hiddens);
    Ok(())
}

#[pyfunction]
#[pyo3(signature = (states, n_bins=16))]
fn compute_phi(states: PyReadonlyArray2<f32>, n_bins: usize) -> PyResult<(f64, f64, f64)> {
    let array = states.as_array();
    let n_cells = array.nrows();
    let hiddens: Vec<Vec<f32>> = (0..n_cells)
        .map(|i| array.row(i).to_vec())
        .collect();
    let hrefs: Vec<&[f32]> = hiddens.iter().map(|h| h.as_slice()).collect();
    let (phi, comp) = anima_core::phi_iit(&hrefs, n_bins);
    Ok((phi, comp.total_mi, comp.min_partition_mi))
}

fn anima_rs(m: &Bound<'_, PyModule>) -> PyResult<()> {
    m.add("__version__", "0.1.0")?;

    // Talk5 submodule
    let talk5 = PyModule::new(m.py(), "talk5")?;
    talk5.add_function(wrap_pyfunction!(talk5_run, &talk5)?)?;
    talk5.add_function(wrap_pyfunction!(talk5_get_hiddens, &talk5)?)?;
    talk5.add_function(wrap_pyfunction!(talk5_set_hiddens, &talk5)?)?;
    m.add_submodule(&talk5)?;

    // Alpha sweep submodule
    let sweep = PyModule::new(m.py(), "alpha_sweep")?;
    sweep.add_function(wrap_pyfunction!(alpha_sweep_run, &sweep)?)?;
    m.add_submodule(&sweep)?;

    // Golden MoE submodule
    let moe = PyModule::new(m.py(), "golden_moe")?;
    moe.add_function(wrap_pyfunction!(golden_moe_forward, &moe)?)?;
    m.add_submodule(&moe)?;

    // Transplant submodule
    let tx = PyModule::new(m.py(), "transplant")?;
    tx.add_function(wrap_pyfunction!(transplant_run, &tx)?)?;
    m.add_submodule(&tx)?;

    // Phi backward compat
    m.add_function(wrap_pyfunction!(compute_phi, m)?)?;

    Ok(())
}
